#wifi managing on Ubuntu:

    ##/etc/network/interfaces:
            allow-hotplug wlan0
            iface wlan0 inet dhcp
                wpa-ssid <SSID>
                wpa-psk  <PASSWORD>

    ## or in case you need to have multiple entries:
            allow-hotplug wlan0
            iface wlan0 inet dhcp
            iface work inet dhcp
                wpa-ssid <SSID_WORK>
                wpa-psk  <PASSWORD_WORK>

            iface home inet dhcp
                wpa-ssid <SSID_HOME>
                wpa-psk  <PASSWORD_HOME>

    ### then to use either:
            ifup wlan0=work
        or
            ifup wlan0=home

    ## to turn off:
            ifdown wla
